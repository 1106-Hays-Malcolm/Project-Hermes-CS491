        <script>
            function onSubmit() {
                const xhttp = new XMLHttpRequest(); 
                xhttp.open("POST", "/form-submit", true);
                xhttp.setRequestHeader("Content-type", "application/json");

                var mission_name = document.getElementById("mission-name").value;
                var question = document.getElementById("question").value;

                xhttp.send(JSON.stringify({"mission-name": mission_name, "question": question}));

                document.getElementById("question").value = "";

                if (question != "") {
                    const newP = document.createElement("p");
                    const newP2 = document.createElement("pre");

                    const newContent = document.createTextNode("You: " + question);
                    const newContent2 = document.createTextNode("Hermes:");

                    document.getElementById("context-container").appendChild(newP);
                    document.getElementById("context-container").appendChild(newP2);

                    newP.appendChild(newContent);
                    newP2.appendChild(newContent2);
                }
            }

            function updateNeedles(degrees) {
                var needles = document.getElementsByClassName("compass-needle");

                for (var i = 0; i < needles.length; i++) {
                    needles[i].style.transform = "rotate(" + degrees + "deg)";
                }
            }

            // constant compass update call
            (function(){
                const xhttp = new XMLHttpRequest(); 

                xhttp.onload = function() {
                    var degrees = parseFloat(this.responseText);
                    console.log("Compass Update: " + degrees);

                    updateNeedles(degrees);
                }

                xhttp.open("GET", "/update-compass");
                xhttp.send();

                setTimeout(arguments.callee, 500);
            })();

            // Get conversation update call
            (function(){
                const xhttp = new XMLHttpRequest(); 

                xhttp.onload = function() {
                    var response = JSON.parse(this.responseText);
                    console.log(response);

                    var context = "";

                    if (response != "") {
                        response.forEach(function(item, index) {
                            console.log("loop: " + item);
                            context += item;
                        });

                        responseArea = document.getElementById("context-container").lastChild;
                        responseArea.innerHTML += context;
                    }
                }

                xhttp.open("GET", "/get-conversation");
                xhttp.send();

                setTimeout(arguments.callee, 500);
            })();

        </script>
